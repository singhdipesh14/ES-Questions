  AREA RESET, DATA, READONLY
    EXPORT __Vectors
__Vectors
  DCD 0X10001000
  DCD Reset_Handler
  
  AREA mycode, CODE, READONLY
  ENTRY
  EXPORT Reset_Handler
Reset_Handler
  MOV R0, #10;NUMERATOR, WILL BECOME REMAINDER LATER
  MOV R1, #3;DENOMINATOR
  MOV R2, #0; QUOTIENT
  LDR R4, =QUO
  LDR R5, =REM
 UP CMP R0, R1
  BCC/BLO STORE
  SUBS R0, R1
  ADD R2,#1
  B UP

STORE STR R2, [R4]
  STR R0, [R5]

STOP B STOP

  AREA myData, DATA, READWRITE
QUO DCD 0X0
REM DCD 0X0
  END
